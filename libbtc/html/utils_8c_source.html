47 {
48  int bLen = inLen / 2;

51  int i;
52  for (i = 0; i < bLen; i++) {
53  c = 0;
54  if (str[i * 2] >= '0' && str[i * 2] <= '9') {
55  *out = (str[i * 2] - '0') << 4;
56  }

59  }

62  }
63  if (str[i * 2 + 1] >= '0' && str[i * 2 + 1] <= '9') {
64  *out |= (str[i * 2 + 1] - '0');
65  }
66  if (str[i * 2 + 1] >= 'a' && str[i * 2 + 1] <= 'f') {

157  if (i < 0xfd) {
